実装ポイント

  • プリセットは直接今日の記録へ出さず、マイ種目化してから選択対象にする。
  • 追加済みプリセットは `Exercise.sourcePresetId` で判定し、再追加不可にする。
  • マイ種目の削除は物理削除ではなく `isActive: false`。
  • 最近使った種目は `WorkoutExercise.createdAt` から派生算出する。

3. 重要コード

export async function addFromPresets(presetIds: UUID[]) {
  return db.transaction('rw', db.exercisePresets, db.exercises, async () => {
    const existing = await db.exercises
      .where('sourcePresetId')
      .anyOf(presetIds)
      .toArray();
    const existingIds = new Set(existing.map((exercise) => exercise.sourcePresetId));
    const presets = await db.exercisePresets.where('id').anyOf(presetIds).toArray();

    const nextSortOrder = await getNextExerciseSortOrder();
    const exercises = presets
      .filter((preset) => !existingIds.has(preset.id))
      .map((preset, index) => createExerciseFromPreset(preset, nextSortOrder + index * 10));

    await db.exercises.bulkAdd(exercises);
    return exercises;
  });
}
