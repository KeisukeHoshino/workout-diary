実装ポイント

  • メニュー名は必須。メモは任意。
  • メニュー内種目は `MenuTemplateExercise.sortOrder` で保持する。
  • 今日へ追加すると `WorkoutExercise` と空の `WorkoutSet` を生成する。
  • 既存カードがある場合はカードを増やさず空セットを追加する。

2. データフロー

MenuTemplateユーザーが作成した1回分メニュー。
MenuTemplateExercisesortOrder 昇順で取得。
WorkoutExercise既存カードがなければ作成。
WorkoutSet空の1セット目を作成。
画面更新WorkoutDetail を再取得。

3. 重要コード

export async function addMenuToDate(date: LocalDateString, menuTemplateId: UUID) {
  await db.transaction('rw', db.workoutDays, db.workoutExercises, db.workoutSets, db.menuTemplateExercises, async () => {
    const day = await ensureWorkoutDay(date);
    const items = await db.menuTemplateExercises
      .where({ menuTemplateId })
      .sortBy('sortOrder');

    for (const item of items) {
      const existing = await db.workoutExercises.where({ workoutDayId: day.id, exerciseId: item.exerciseId }).first();
      if (existing) {
        await db.workoutSets.add(createEmptyWorkoutSet(existing.id, await nextSetNumber(existing.id)));
        continue;
      }
      const workoutExercise = createWorkoutExercise({ workoutDayId: day.id, exerciseId: item.exerciseId, sortOrder: await nextWorkoutExerciseSortOrder(day.id) });
      await db.workoutExercises.add(workoutExercise);
      await db.workoutSets.add(createEmptyWorkoutSet(workoutExercise.id, 1));
    }
  });
  return getWorkoutByDate(date);
}
