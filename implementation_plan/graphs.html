export function aggregateMaxWeightPoints(rows: MaxWeightSourceRow[]): MaxWeightPoint[] {
  const byDate = new Map<LocalDateString, MaxWeightPoint>();

  for (const row of rows) {
    if (row.set.weightKg === null) continue;
    const current = byDate.get(row.date);
    const candidate = toMaxWeightPoint(row);

    const shouldReplace =
      !current ||
      candidate.weightKg > current.weightKg ||
      (candidate.weightKg === current.weightKg && (candidate.reps ?? 0) > (current.reps ?? 0));

    if (shouldReplace) byDate.set(row.date, candidate);
  }

  return [...byDate.values()].sort((a, b) => a.date.localeCompare(b.date));
}
